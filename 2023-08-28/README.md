**Q14 구조체 (정보처리기사 22년 1회)**
````
#include <stdio.h>  // 표준 입력과 출력 관련 함수를 사용하기 위한 라이브러리 포함

struct student {   // student라는 이름의 구조체 정의
    int n, g;      // 구조체 내부에 정수형 변수 n과 g 선언
};

int main() {       // 프로그램의 시작점

    struct student st[2];  // student 구조체 타입을 가지는 배열 st를 선언 (크기: 2)
    int i = 0;             // 반복문에서 사용할 정수형 변수 i 초기화

    for (i; i < 2; i++) {  // i가 0부터 1까지 반복
        st[i].n = i;       // st[i]의 n 멤버에 i 값을 할당
        st[i].g = i + 1;   // st[i]의 g 멤버에 (i + 1) 값을 할당
    }

    printf("%d", st[0].n + st[1].g);  // st[0]의 n 값과 st[1]의 g 값의 합을 출력
                                     // 출력 예: 0 + (1 + 1) = 2

    return 0;  // 프로그램 종료를 나타내는 값 0을 반환
}
````

**Q15 문자열 포인터 (정보처리기사 22년 2회)**
````
#include <stdio.h>  // 표준 입력과 출력 관련 함수를 사용하기 위한 라이브러리 포함

int len(char *p);  // len 함수의 프로토타입 선언

int main() {       // 프로그램의 시작점

    char *p1 = "2022";    // 문자열 "2022"의 시작 주소를 가리키는 포인터 p1 선언
    char *p2 = "202207";  // 문자열 "202207"의 시작 주소를 가리키는 포인터 p2 선언
    
    int a = p1;  // 포인터 p1의 주소를 정수형 변수 a에 할당 (주소를 정수로 변환)
    int b = p2;  // 포인터 p2의 주소를 정수형 변수 b에 할당 (주소를 정수로 변환)
    
    printf("%d", len(a) + len(b));  // len 함수로 구한 문자열 길이의 합을 출력 // 4+6=10 출력
    
    return 0;  // 프로그램 종료를 나타내는 값 0을 반환
}

int len(char *p) {  // 문자열의 길이를 반환하는 함수 정의
    int r = 0;  // 결과 값을 저장할 변수 r 초기화
    while (*p != '\0') {  // 문자열의 끝을 나타내는 널 문자('\0')를 만날 때까지 반복
        p++;   // 다음 문자의 주소로 포인터 이동
        r++;   // 길이 값 증가
    }
    return r;  // 문자열의 길이 반환
}

#include <stdio.h>  // 표준 입력과 출력 관련 함수를 사용하기 위한 라이브러리 포함

int main(int argc, char *argv[]) {  // 프로그램의 시작점, 명령줄 인수를 사용하는 형태로 main 함수 정의
    int a[4] = {0, 2, 4, 8};  // 정수형 배열 a 선언 및 초기화
    int b[3] = {};  // 정수형 배열 b 선언 및 초기화 (모든 요소는 0)
    int i = 1;  // 반복문에서 사용할 정수형 변수 i 초기화
    int sum = 0;  // 합을 저장할 정수형 변수 sum 초기화
    int *p1;  // 정수형 포인터 p1 선언

    for (i; i < 4; i++) {  // i가 1부터 3까지 반복
        p1 = a + i;  // p1에 a[i]의 주소를 할당
        b[i - 1] = *p1 - a[i - 1];  // b[i-1]에 *(a + i) - a[i - 1] 값을 할당
        sum = sum + b[i - 1] + a[i];  // sum에 이전 sum 값과 b[i-1] + a[i] 값을 더한 후 할당
    }

    printf("%d", sum);  // sum 값을 출력__22 출력
    
    return 0;  // 프로그램 종료를 나타내는 값 0을 반환
}

**Q17 (정보처리기사 22년 3회)**
````
#include <stdio.h>  // 표준 입력과 출력 관련 함수를 사용하기 위한 라이브러리 포함

int main() {  // 프로그램의 시작점

    int n;  // 정수 n 선언
    int k;  // 정수 k 선언
    int s;  // 정수 s 선언
    int el = 0;  // 완전수의 개수를 저장하는 정수 el 초기화
    
    for (n = 6; n <= 30; n++) {  // n이 6부터 30까지 반복
        s = 0;  // s 초기화
        k = n / 2;  // k를 n의 절반으로 초기화
        for (int j = 1; j <= k; j++) {  // j가 1부터 k까지 반복
            if (n % j == 0) {  // n이 j로 나누어 떨어진다면
                s = s + j;  // s에 j를 더함
            }
        }
        if (s == n) {  // s가 n과 같다면 (완전수인지 확인)
            el++;  // 완전수의 개수 증가
        }
    }
  
    printf("%d", el);  // 완전수의 개수를 출력
    
    return 0;  // 프로그램 종료를 나타내는 값 0을 반환
}
